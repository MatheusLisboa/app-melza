const fmt = (n: number, compact = false) => {
  if (compact && n >= 1000) {
    return `R$ ${(n / 1000).toFixed(1).replace(".", ",")}k`;
  }
  return new Intl.NumberFormat("pt-BR", { style: "currency", currency: "BRL" }).format(n);
};

function Btn({
  children, variant = "primary", size = "md", wsColor, onClick, fullWidth, icon, disabled,
}: {
  children: React.ReactNode;
  variant?: "primary" | "secondary" | "ghost" | "destructive";
  size?: "sm" | "md" | "lg";
  wsColor?: string;
  onClick?: () => void;
  fullWidth?: boolean;
  icon?: React.ReactNode;
  disabled?: boolean;
}) {
  const base = "inline-flex items-center justify-center gap-2 rounded-xl font-medium transition-all active:scale-[0.97] select-none cursor-pointer";
  const sizes = { sm: "h-8 px-3 text-xs", md: "h-11 px-4 text-sm", lg: "h-[52px] px-5 text-[15px]" };
  const variants = {
    primary: `text-white`,
    secondary: "bg-white/[0.07] text-white/80 hover:bg-white/[0.11]",
    ghost: "text-white/50 hover:text-white/80 hover:bg-white/[0.06]",
    destructive: "bg-[#EF444422] text-[#EF4444] hover:bg-[#EF444433]",
  };
  const bg = variant === "primary" ? (wsColor || "#6366F1") : undefined;
  return (
    <button
      className={`${base} ${sizes[size]} ${variants[variant]} ${fullWidth ? "w-full" : ""} ${disabled ? "opacity-40 pointer-events-none" : ""}`}
      style={bg ? { backgroundColor: bg } : undefined}
      onClick={onClick}
    >
      {icon && <span className="shrink-0">{icon}</span>}
      {children}
    </button>
  );
}

function InputField({
  label, type = "text", placeholder, value, onChange, icon, rightEl, hint,
}: {
  label?: string;
  type?: string;
  placeholder?: string;
  value?: string;
  onChange?: (v: string) => void;
  icon?: React.ReactNode;
  rightEl?: React.ReactNode;
  hint?: string;
}) {
  return (
    <div className="flex flex-col gap-1.5">
      {label && <label className="text-xs font-medium text-white/50 tracking-wide">{label}</label>}
      <div className="relative flex items-center">
        {icon && <div className="absolute left-3.5 text-white/30">{icon}</div>}
        <input
          type={type}
          placeholder={placeholder}
          value={value}
          onChange={(e) => onChange?.(e.target.value)}
          className="w-full h-[50px] bg-[#18181B] border border-white/[0.08] rounded-xl text-white/90 placeholder:text-white/25 focus:outline-none focus:border-white/20 transition-colors"
          style={{ paddingLeft: icon ? 44 : 16, paddingRight: rightEl ? 44 : 16 }}
        />
        {rightEl && <div className="absolute right-3.5 text-white/40">{rightEl}</div>}
      </div>
      {hint && <span className="text-xs text-white/30 px-1">{hint}</span>}
    </div>
  );
}

function MoneyDisplay({ amount, size = "xl", color, wsColor }: { amount: number; size?: "sm" | "md" | "xl" | "2xl"; color?: string; wsColor?: string }) {
  const sizes = { sm: "text-lg", md: "text-2xl", xl: "text-[34px]", "2xl": "text-[42px]" };
  const [int, dec] = Math.abs(amount).toFixed(2).split(".");
  const formatted = parseInt(int).toLocaleString("pt-BR");
  const textColor = color || "rgba(255,255,255,0.95)";
  return (
    <span className={`${sizes[size]} font-semibold tracking-tight leading-none`} style={{ ...mono, color: textColor }}>
      <span className="text-[0.55em] mr-0.5 opacity-60">R$</span>
      {formatted}
      <span className="text-[0.65em] opacity-60">,{dec}</span>
    </span>
  );
}

// Attribution trio used in transaction items
function AttributionTrio({ consumer, payer, cardOwner }: { consumer: Member; payer: Member; cardOwner: Member }) {
  const allSame = consumer.id === payer.id && payer.id === cardOwner.id;
  if (allSame) return null;

  const fields = [
    { label: "consumiu", member: consumer, show: true },
    { label: "pagou", member: payer, show: payer.id !== consumer.id },
    { label: "cartão", member: cardOwner, show: cardOwner.id !== payer.id },
  ].filter(f => f.show);

  return (
    <div className="flex items-center gap-2 mt-1">
      {fields.map(({ label, member }, i) => (
        <div key={i} className="flex items-center gap-1">
          {i > 0 && <span className="text-white/15 text-[10px]">·</span>}
          <span className="text-[10px] text-white/30">{label}</span>
          <div className="w-3.5 h-3.5 rounded-full flex items-center justify-center text-[7px] font-bold text-white shrink-0" style={{ backgroundColor: member.color }}>
            {member.initials[0]}
          </div>
          <span className="text-[10px] text-white/50 font-medium">{member.name}</span>
        </div>
      ))}
    </div>
  );
}

        {tx.type === "income" ? "+" : "−"}{fmt(tx.amount)}

    </button>
  );

function EmailConfirmScreen({ nav }: { nav: (s: ScreenId) => void }) {
  const [code, setCode] = useState(["", "", "", "", "", ""]);
  const refs = useRef<(HTMLInputElement | null)[]>([]);

  const handleChange = (i: number, val: string) => {
    if (!/^\d*$/.test(val)) return;
    const next = [...code];
    next[i] = val.slice(-1);
    setCode(next);
    if (val && i < 5) refs.current[i + 1]?.focus();
  };

  const handleKey = (i: number, e: React.KeyboardEvent) => {
    if (e.key === "Backspace" && !code[i] && i > 0) refs.current[i - 1]?.focus();
  };

  const filled = code.every(c => c !== "");

  return (
    <div className="flex flex-col h-full bg-[#09090B] px-6">
      <StatusBar />
      <div className="flex items-center gap-3 py-3">
        <button onClick={() => nav("signup")} className="w-8 h-8 flex items-center justify-center rounded-xl bg-white/[0.06]">
          <ChevronLeft size={18} strokeWidth={2} className="text-white/70" />
        </button>
      </div>

      <div className="flex-1 flex flex-col gap-8 pt-6 pb-8">
        <div>
          <h2 className="text-white text-[26px] font-semibold tracking-tight">Confirme seu e-mail</h2>
          <p className="text-white/35 text-sm mt-1.5">Enviamos um código de 6 dígitos para<br /><span className="text-white/55">matheus@email.com</span></p>
        </div>

        <div className="flex gap-2.5 justify-center">
          {code.map((digit, i) => (
            <input
              key={i}
              ref={el => { refs.current[i] = el; }}
              type="text"
              inputMode="numeric"
              maxLength={1}
              value={digit}
              onChange={e => handleChange(i, e.target.value)}
              onKeyDown={e => handleKey(i, e)}
              className="w-12 h-14 rounded-xl bg-[#18181B] border border-white/[0.08] text-white text-xl font-bold text-center focus:outline-none focus:border-[#6366F1]/60 transition-colors"
              style={mono}
            />
          ))}
        </div>

        <div className="mt-auto flex flex-col gap-3">
          <Btn variant="primary" size="lg" fullWidth disabled={!filled} onClick={() => nav("onboarding")}>
            Confirmar
          </Btn>
          <p className="text-center text-sm text-white/30">
            Não recebeu?{" "}
            <button className="text-white/55 font-medium hover:text-white/80 transition-colors">
              Reenviar
            </button>
          </p>
        </div>
      </div>
    </div>
  );
}

// ─────────────────────────────────────────────────────────
// TRANSACTIONS
// ─────────────────────────────────────────────────────────

function TransactionsScreen({ nav, ws }: { nav: (s: ScreenId) => void; ws: WorkspaceData }) {
  const [filter, setFilter] = useState<"all" | "income" | "expense">("all");
  const [search, setSearch] = useState("");

  const filters: { id: typeof filter; label: string }[] = [
    { id: "all", label: "Tudo" },
    { id: "income", label: "Receitas" },
    { id: "expense", label: "Despesas" },
  ];

  const filtered = TRANSACTIONS.filter(tx => {
    if (filter !== "all" && tx.type !== filter) return false;
    if (search && !tx.title.toLowerCase().includes(search.toLowerCase())) return false;
    return true;
  });

  const grouped = filtered.reduce<Record<string, Transaction[]>>((acc, tx) => {
    if (!acc[tx.dateGroup]) acc[tx.dateGroup] = [];
    acc[tx.dateGroup].push(tx);
    return acc;
  }, {});

  return (
    <div className="flex flex-col h-full bg-[#09090B] overflow-hidden">
      <StatusBar />
      <TopBar title="Histórico" rightEl={
        <button className="w-9 h-9 flex items-center justify-center rounded-xl bg-white/[0.06]">
          <SlidersHorizontal size={16} strokeWidth={2} className="text-white/60" />
        </button>
      } />

      {/* Search */}
      <div className="px-5 pb-3 shrink-0">
        <div className="relative flex items-center">
          <Search size={15} className="absolute left-3.5 text-white/30" />
          <input
            type="text"
            placeholder="Buscar transações..."
            value={search}
            onChange={e => setSearch(e.target.value)}
            className="w-full h-[42px] bg-[#18181B] border border-white/[0.07] rounded-xl text-white/80 text-sm placeholder:text-white/25 focus:outline-none focus:border-white/15 transition-colors"
            style={{ paddingLeft: 38 }}
          />
        </div>
      </div>

      {/* Filter chips */}
      <div className="flex gap-2 px-5 pb-4 shrink-0">
        {filters.map(f => (
          <button
            key={f.id}
            onClick={() => setFilter(f.id)}
            className="px-3.5 py-1.5 rounded-full text-[13px] font-medium transition-all"
            style={{
              background: filter === f.id ? `${ws.color}25` : "#18181B",
              color: filter === f.id ? ws.color : "rgba(255,255,255,0.4)",
              border: `1px solid ${filter === f.id ? `${ws.color}45` : "transparent"}`,
            }}
          >
            {f.label}
          </button>
        ))}
      </div>

      <div className="flex-1 overflow-y-auto pb-[90px] px-5">
        {Object.entries(grouped).length === 0 ? (
          <EmptyState
            icon={<ArrowUpDown size={24} />}
            title="Nenhuma transação"
            desc="Adicione sua primeira transação para começar a organizar suas finanças."
            cta="Nova transação"
            onCta={() => nav("transaction-new")}
            wsColor={ws.color}
          />
        ) : (
          Object.entries(grouped).map(([group, txs]) => (
            <div key={group} className="mb-4">
              <p className="text-xs font-semibold text-white/35 uppercase tracking-wider mb-1 py-1">{group}</p>
              {txs.map(tx => (
                <TxRow key={tx.id} tx={tx} onTap={() => nav("transaction-detail")} wsColor={ws.color} />
              ))}
            </div>
          ))
        )}
      </div>

      <BottomNav active="transactions" onNavigate={nav} wsColor={ws.color} />
      <FAB onPress={() => nav("transaction-new")} wsColor={ws.color} />
    </div>
  );
}

function NewTransactionScreen({ nav, ws }: { nav: (s: ScreenId) => void; ws: WorkspaceData }) {
  const [type, setType] = useState<"expense" | "income">("expense");
  const [amount, setAmount] = useState("");
  const [title, setTitle] = useState("");
  const [consumer, setConsumer] = useState(MATHEUS);
  const [payer, setPayer] = useState(MATHEUS);
  const [cardOwner, setCardOwner] = useState(MATHEUS);
  const [selectedCat, setSelectedCat] = useState("Alimentação");

  const categories = [
    { id: "Alimentação", emoji: "🛒" }, { id: "Transporte", emoji: "⛽" },
    { id: "Moradia", emoji: "🏠" }, { id: "Saúde", emoji: "💊" },
    { id: "Lazer", emoji: "🎬" }, { id: "Receita", emoji: "💼" },
    { id: "Streaming", emoji: "📺" }, { id: "Outros", emoji: "📦" },
  ];

  const members = ws.members.length > 1 ? ws.members : [MATHEUS, ANA];

  const handleAmount = (v: string) => {
    const clean = v.replace(/\D/g, "");
    if (clean.length > 9) return;
    const num = parseInt(clean || "0") / 100;
    setAmount(num === 0 ? "" : num.toFixed(2).replace(".", ",").replace(/\B(?=(\d{3})+(?!\d))/g, "."));
  };

  return (
    <div className="flex flex-col h-full bg-[#09090B]">
      <StatusBar />
      <div className="flex items-center justify-between px-5 py-3 shrink-0">
        <button onClick={() => nav("transactions")} className="w-8 h-8 flex items-center justify-center rounded-xl bg-white/[0.06]">
          <X size={16} strokeWidth={2} className="text-white/70" />
        </button>
        <span className="text-[15px] font-semibold text-white/90">Nova transação</span>
        <div className="w-8" />
      </div>

      {/* Type toggle */}
      <div className="flex gap-2 px-5 pb-4 shrink-0">
        {(["expense", "income"] as const).map(t => (
          <button
            key={t}
            onClick={() => setType(t)}
            className="flex-1 py-2.5 rounded-xl text-[13px] font-semibold transition-all"
            style={{
              background: type === t ? (t === "expense" ? "#EF444420" : "#22C55E20") : "#18181B",
              color: type === t ? (t === "expense" ? "#EF4444" : "#22C55E") : "rgba(255,255,255,0.35)",
            }}
          >
            {t === "expense" ? "Despesa" : "Receita"}
          </button>
        ))}
      </div>

      <div className="flex-1 overflow-y-auto pb-8">
        {/* Amount */}
        <div className="flex flex-col items-center py-6 px-5 gap-2">
          <p className="text-xs text-white/35 font-medium uppercase tracking-wider">Valor</p>
          <div className="flex items-center gap-2">
            <span className="text-xl text-white/40 font-semibold">R$</span>
            <input
              type="text"
              inputMode="numeric"
              placeholder="0,00"
              value={amount}
              onChange={e => handleAmount(e.target.value)}
              className="text-[42px] font-bold text-white bg-transparent border-none outline-none w-48 text-center placeholder:text-white/15"
              style={mono}
            />
          </div>
        </div>

        <div className="px-5 flex flex-col gap-4">
          <InputField label="Descrição" placeholder="Ex: Supermercado Extra" value={title} onChange={setTitle} />

          {/* Categories */}
          <div>
            <label className="text-xs text-white/40 font-medium uppercase tracking-wider">Categoria</label>
            <div className="flex gap-2 mt-2 overflow-x-auto pb-1 -mx-1 px-1 scrollbar-none" style={{ scrollbarWidth: "none" }}>
              {categories.map(({ id, emoji }) => (
                <button
                  key={id}
                  onClick={() => setSelectedCat(id)}
                  className="flex flex-col items-center gap-1 shrink-0 w-[62px]"
                >
                  <div
                    className="w-12 h-12 rounded-xl flex items-center justify-center text-xl transition-all"
                    style={{
                      background: selectedCat === id ? `${ws.color}25` : "#18181B",
                      border: `1.5px solid ${selectedCat === id ? ws.color : "transparent"}`,
                    }}
                  >
                    {emoji}
                  </div>
                  <span className="text-[10px] text-white/35 text-center leading-tight">{id}</span>
                </button>
              ))}
            </div>
          </div>

          {/* Attribution — the key feature */}
          <div>
            <label className="text-xs text-white/40 font-medium uppercase tracking-wider mb-3 block">Atribuição</label>
            <div className="rounded-2xl bg-[#111113] border border-white/[0.06] overflow-hidden">
              {[
                { label: "Quem consumiu?", value: consumer, onChange: setConsumer, desc: "Quem se beneficiou" },
                { label: "Quem pagou?", value: payer, onChange: setPayer, desc: "Quem desembolsou" },
                { label: "Dono do cartão?", value: cardOwner, onChange: setCardOwner, desc: "Qual cartão foi usado" },
              ].map(({ label, value, onChange, desc }, i) => (
                <div key={i}>
                  <div className="px-4 py-3.5 flex items-center gap-3">
                    <div className="flex-1">
                      <p className="text-[13px] font-medium text-white/80">{label}</p>
                      <p className="text-[11px] text-white/30 mt-0.5">{desc}</p>
                    </div>
                    <div className="flex gap-1.5">
                      {members.map(m => (
                        <button
                          key={m.id}
                          onClick={() => onChange(m)}
                          className="flex items-center gap-1 px-2 py-1.5 rounded-lg transition-all"
                          style={{
                            background: value.id === m.id ? `${m.color}22` : "#1C1C1F",
                            border: `1.5px solid ${value.id === m.id ? m.color : "transparent"}`,
                          }}
                        >
                          <div className="w-4 h-4 rounded-full flex items-center justify-center text-[8px] font-bold text-white" style={{ backgroundColor: m.color }}>
                            {m.initials[0]}
                          </div>
                          <span className="text-[11px] font-medium" style={{ color: value.id === m.id ? m.color : "rgba(255,255,255,0.4)" }}>
                            {m.name}
                          </span>
                        </button>
                      ))}
                    </div>
                  </div>
                  {i < 2 && <Divider />}
                </div>
              ))}
            </div>
          </div>

          <Btn
            variant="primary"
            size="lg"
            fullWidth
            wsColor={ws.color}
            disabled={!amount || !title}
            onClick={() => nav("transactions")}
          >
            Salvar transação
          </Btn>
        </div>
      </div>
    </div>
  );
}

    { id: "2", role: "user", text: "Quanto gastei com alimentação este mês?" },
    { id: "3", role: "assistant", text: "Você gastou **R$ 1.240,50** com alimentação em julho, distribuídos em 8 transações. Esse valor representa 38% das suas despesas totais no mês. Comparado a junho, você gastou 12% mais." },
  ]);

  const suggestions = [
    "Qual meu saldo atual?",
    "Resumo de julho",
    "Gastos com cartão",
    "Quem me deve dinheiro?",
  ];

  const send = () => {
    if (!input.trim()) return;
    setMessages(prev => [
      ...prev,
      { id: Date.now().toString(), role: "user", text: input },
      { id: (Date.now() + 1).toString(), role: "assistant", text: "Deixa eu analisar isso para você... ✨" },

                </button>
              ))}
            </div>
          </div>
        )}

        {/* Messages */}
        <div className="flex flex-col gap-3">
          {messages.map(msg => (
            <div key={msg.id} className={`flex gap-2.5 ${msg.role === "user" ? "flex-row-reverse" : ""}`}>
              {msg.role === "assistant" && (
                <div className="w-7 h-7 rounded-full flex items-center justify-center shrink-0 mt-1" style={{ background: `${ws.color}25` }}>
                  <Sparkles size={12} strokeWidth={2} style={{ color: ws.color }} />

                  borderBottomLeftRadius: msg.role === "assistant" ? 6 : undefined,
                }}
              >
                {msg.text.replace(/\*\*(.*?)\*\*/g, "$1")}
              </div>
            </div>
          ))}
        </div>
      </div>

      {/* Input */}
      <div className="px-4 pb-5 pt-3 border-t border-white/[0.06] shrink-0" style={{ background: "rgba(9,9,11,0.95)", backdropFilter: "blur(20px)" }}>

            value={input}
            onChange={e => setInput(e.target.value)}

            className="flex-1 h-11 bg-[#18181B] border border-white/[0.07] rounded-xl text-white/80 text-sm placeholder:text-white/25 focus:outline-none focus:border-white/15 transition-colors px-4"

export default function App() {
  const [screen, setScreen] = useState<ScreenId>("login");
  const [wsIndex, setWsIndex] = useState(0);
  const ws = WORKSPACES[wsIndex];

  const nav = (s: ScreenId) => setScreen(s);

    const i = WORKSPACES.findIndex(x => x.id === w.id);

  };

  const renderScreen = () => {
    switch (screen) {
      case "login": return <LoginScreen nav={nav} />;
      case "signup": return <SignupScreen nav={nav} />;
      case "forgot-password": return <ForgotScreen nav={nav} />;
      case "email-confirm": return <EmailConfirmScreen nav={nav} />;
      case "onboarding": return <OnboardingScreen nav={nav} />;
      case "workspace-switcher": return <WorkspaceSwitcher current={ws} nav={nav} onSelect={selectWs} />;
      case "workspace-create": return <CreateWorkspaceScreen nav={nav} />;
      case "workspace-members": return <WorkspaceMembersScreen nav={nav} ws={ws} />;
      case "workspace-edit": return <WorkspaceMembersScreen nav={nav} ws={ws} />;
      case "dashboard": return <PersonalDashboard nav={nav} ws={ws} onSwitchWs={() => nav("workspace-switcher")} />;
      case "dashboard-shared": return <SharedDashboard nav={nav} ws={WORKSPACES[1]} />;
      case "cards": return <CardsScreen nav={nav} ws={ws} />;
      case "card-detail": return <CardDetailScreen nav={nav} ws={ws} />;
      case "card-new": return <CreateWorkspaceScreen nav={nav} />;
      case "accounts": return <AccountsScreen nav={nav} ws={ws} />;
      case "account-new": return <AccountsScreen nav={nav} ws={ws} />;
      case "transactions": return <TransactionsScreen nav={nav} ws={ws} />;
      case "transaction-detail": return <TransactionDetail nav={nav} ws={ws} />;
      case "transaction-new": return <NewTransactionScreen nav={nav} ws={ws} />;
      case "subscriptions": return <SubscriptionsScreen nav={nav} ws={ws} />;
      case "loans": return <LoansScreen nav={nav} ws={ws} />;
      case "entre-nos": return <EntreNosScreen nav={nav} ws={ws} />;
      case "chat": return <ChatScreen nav={nav} ws={ws} />;
      case "profile": return <ProfileScreen nav={nav} ws={ws} />;
      case "settings": return <SettingsScreen nav={nav} ws={ws} />;
      default: return <LoginScreen nav={nav} />;
    }
  };

  return (
    <div className="min-h-screen bg-[#050507] flex flex-col items-center justify-start py-6 px-4 gap-6">
      {/* Background glow */}
      <div className="fixed inset-0 pointer-events-none">
        <div className="absolute top-0 left-1/2 -translate-x-1/2 w-[600px] h-[300px] rounded-full opacity-[0.06] blur-3xl" style={{ background: ws.color }} />
      </div>

      {/* App identity */}
      <div className="flex items-center gap-3 relative z-10">
        <div className="w-8 h-8 rounded-xl flex items-center justify-center" style={{ background: "linear-gradient(135deg, #6366F1, #8B5CF6)" }}>
          <span className="text-white text-sm font-bold" style={{ letterSpacing: "-0.04em" }}>N</span>
        </div>
        <span className="text-white/60 text-sm font-semibold tracking-tight">Nivo</span>
        <span className="text-white/20 text-sm">·</span>
        <span className="text-white/30 text-xs">Design Prototype</span>
        <div className="flex items-center gap-1.5 px-2.5 py-1 rounded-full ml-2" style={{ background: `${ws.color}20`, border: `1px solid ${ws.color}35` }}>
          <span className="text-sm">{ws.emoji}</span>
          <span className="text-[11px] font-semibold" style={{ color: ws.color }}>{ws.name}</span>
        </div>
      </div>

      {/* Screen navigator */}
      <div className="flex flex-wrap gap-3 justify-center max-w-[900px] relative z-10">
        {SCREEN_GROUPS.map(({ label, screens }) => (
          <div key={label} className="flex items-center gap-1 p-1 rounded-xl bg-white/[0.04] border border-white/[0.06]">
            <span className="text-[9px] text-white/25 font-semibold uppercase tracking-wider px-2">{label}</span>
            {screens.map(({ id, label: sl }) => (
              <button
                key={id}
                onClick={() => nav(id)}
                className="px-3 py-1.5 rounded-lg text-[11px] font-medium transition-all"
                style={{
                  background: screen === id ? `${ws.color}25` : "transparent",
                  color: screen === id ? ws.color : "rgba(255,255,255,0.35)",
                  border: `1px solid ${screen === id ? `${ws.color}40` : "transparent"}`,
                }}
              >
                {sl}
              </button>
            ))}
          </div>
        ))}
      </div>

      {/* Phone frame */}
      <div className="relative z-10 flex-shrink-0">
        <div
          className="relative overflow-hidden"
          style={{
            width: 390,
            height: 844,
            borderRadius: 50,
            background: "#09090B",
            boxShadow: "0 40px 120px rgba(0,0,0,0.7), 0 0 0 1px rgba(255,255,255,0.08), inset 0 0 0 1px rgba(255,255,255,0.04)",
          }}
        >
          {/* Side buttons */}
          <div className="absolute -left-[3px] top-28 w-[3px] h-8 rounded-l-full bg-white/10" />
          <div className="absolute -left-[3px] top-44 w-[3px] h-12 rounded-l-full bg-white/10" />
          <div className="absolute -left-[3px] top-60 w-[3px] h-12 rounded-l-full bg-white/10" />
          <div className="absolute -right-[3px] top-40 w-[3px] h-16 rounded-r-full bg-white/10" />
          {/* Screen content */}
          <div className="absolute inset-0 overflow-hidden" style={{ borderRadius: 50 }}>
            {renderScreen()}
          </div>
        </div>
      </div>

      {/* Workspace quick-switch */}
      <div className="flex gap-2 relative z-10">
        {WORKSPACES.map((w, i) => (
          <button
            key={w.id}
            onClick={() => { setWsIndex(i); if (!["login","signup","forgot-password","email-confirm","onboarding"].includes(screen)) setScreen("dashboard"); }}
            className="flex items-center gap-2 px-3 py-2 rounded-xl transition-all text-sm font-medium"
            style={{
              background: wsIndex === i ? `${w.color}20` : "rgba(255,255,255,0.04)",
              border: `1px solid ${wsIndex === i ? `${w.color}40` : "rgba(255,255,255,0.07)"}`,
              color: wsIndex === i ? w.color : "rgba(255,255,255,0.35)",
            }}
          >
            <span>{w.emoji}</span>
            <span className="hidden sm:inline text-[12px]">{w.name}</span>
          </button>
        ))}
      </div>
    </div>
  );
}
